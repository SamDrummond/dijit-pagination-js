dijit-pagination-js

Github page with documentation and more information on how to use this widget.

Example 1

	var Pagination1 = new Pagination({
        totalResults: 100
    }, dojo.byId('pagination1'));
    Pagination1.startup();

Example 2

	var Pagination2 = new Pagination({
        totalResults: 900,
        resultsPerPage: 10,
        currentPage: 1,
        pagesPerSide: 1,
        showPreviousNext: false
    }, dojo.byId('pagination2'));

Example 3

	var Pagination3 = new Pagination({
        totalResults: 225,
        resultsPerPage: 5,
        currentPage: 7,
        pagesPerSide: 3,
        showFirstLast: false
    }, dojo.byId('pagination3'));
    Pagination3.startup();
